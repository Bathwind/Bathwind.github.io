自己的博客也不知道鸽了多久，只记得从六月份打完数学建模后就接着期末考试。期末考试完然后就去接着学车，考驾照，这个月终于要把驾照的事情给忙完了，哈哈哈，开心:happy:，可以接着在家安心宅着写程序了。

一直以来使用pyqt5时候会遇到ui文件反复修改的问题，然后ui文件每次修改的时候导出的Python的程序都会跟着改变，然而以前自己太傻了，每次都重复做着相同的没有意义的事情，把覆盖掉的代码再重新一行一行接着码上，那个过程贼酸爽。:cry:,知道某天突然脑袋灵光了一阵，

想到了既然每次写程序那个Ui转换出来的Py文件都是作为父类来继承，那么咱们可以创建一个新的Py文件，来继承父类的各项东西以及属性，这个没毛病吧。下面举一个例子：

假如上面就是一个ui界面经过Pyuic工具转换后的程序。这个程序就相当于我们的父类程序，我们不要在该程序写自己的方法进去，不然后期ui界面的优化修改会被覆盖掉，（但是如果你是纯手码写界面的话，那就完全不需要这样子）

我们可以新建一个程序上去，然后写一个类来继承父类，程序可以是这个样子：

class myfacewindow(QtWidgets.QMainWindow, Ui_facecollectMainWindow):
    def __init__(self):
        super(myfacewindow, self).__init__()
        self.setupUi(self)
        #定义一个存储名字以及学号的字典
        self.studentdata = {'stu_name':'','stu_id':''}
        #输入框中提示内容
        self.stu_numberlineEdit.setPlaceholderText('输入你的学号')
        self.lineEdit.setPlaceholderText('输入姓名')
        #照片存储路径
        self.path_photos_from_camera = "data/data_faces_from_camera/"
        #opencv以及dlib
        self.cap = cv2.VideoCapture()
        # Dlib 正向人脸检测器 / frontal face detector
        self.detector = dlib.get_frontal_face_detector()
        # 定时器
        self.timer = QTimer(self)
        self.timer.timeout.connect(self.updateFrame)
        # 连接相应的槽函数
        self.isExternalCameraUsed = False#外部摄像头控制选择变量
        self.isFaceDetectEnabled = False
        self.openface_pushButton.setEnabled(False)#未打开摄像头时候人脸识别的按钮是不能被打开的
        self.openface_pushButton.toggled.connect(self.enableFaceDetect)
        self.openface_pushButton.setCheckable(True)
        self.opencamerapushButton_2.toggled.connect(self.startWebcam)#打开摄像头
        self.opencamerapushButton_2.setCheckable(True)
        self.face_showlabel.setText('摄像头未开启')#未打开摄像头时候显示文字摄像头未开启
        self.setFixedSize(self.width(),self.height())#固定窗口尺寸
        self.extern_cameracheckBox.stateChanged.connect(self.useExternalCamera)#选择内置摄像头还是外置摄像头
        #存储数据
        self.storebuttonBox.accepted.connect(self.store_message)
        self.storebuttonBox.rejected.connect(self.clear_message)
    #将输入框中的信息存储起来
    def store_message(self):
        self.studentdata['stu_name'] = self.lineEdit.text()#获得输入框姓名
        self.studentdata['stu_id'] = self.stu_numberlineEdit.text()#学生的学号
        print(self.studentdata)
    #将输入框中的内容全部清空
    def clear_message(self):
        self.lineEdit.setText('')
        self.stu_numberlineEdit.setText('')
    #将拍摄的照片存储起来和设置好多少帧
    # def record_photo(self,fps):
    # 开启/关闭人脸检测
    def enableFaceDetect(self, status):
        if self.cap.isOpened():
            if status:
                self.openface_pushButton.setText('关闭人脸识别')
                self.isFaceDetectEnabled = True
            else:
                self.openface_pushButton.setText('打开人脸识别')
                self.isFaceDetectEnabled = False
        # 检测人脸
    def detectFace(self, frame):
        # 人脸 / Faces
        faces = self.detector(frame, 0)
        font = cv2.FONT_ITALIC
        # 检测到人脸 / Face detected
        if len(faces) != 0:
            # 矩形框 / Show the rectangle box of face
            for k, d in enumerate(faces):
                # 计算矩形大小
                # Compute the width and height of the box
                # (x,y), (宽度width, 高度height)
                pos_start = tuple([d.left(), d.top()])
                pos_end = tuple([d.right(), d.bottom()])
                # 计算矩形框大小 / compute the size of rectangle box
                height = (d.bottom() - d.top())
                width = (d.right() - d.left())
                hh = int(height / 2)
                ww = int(width / 2)
                # 设置颜色 / the color of rectangle of faces detected
                color_rectangle = (255, 255, 255)
                # 判断人脸矩形框是否超出 480x640
                if (d.right() + ww) > 640 or (d.bottom() + hh > 480) or (d.left() - ww < 0) or (d.top() - hh < 0):
                    cv2.putText(frame, "OUT OF RANGE", (20, 300), font, 0.8, (0, 0, 255), 1, cv2.LINE_AA)
                    color_rectangle = (0, 0, 255)
                    print("请调整位置 / Please adjust your position")
                else:
                    color_rectangle = (255, 255, 255)
                    save_flag = 1
                cv2.rectangle(frame,
                              tuple([d.left() - ww, d.top() - hh]),
                              tuple([d.right() + ww, d.bottom() + hh]),
                              color_rectangle, 2)
        return frame
    # 定时器，实时更新画面
    def updateFrame(self):
        ret, frame = self.cap.read()
        if ret:
            self.displayImage(frame)
            if self.isFaceDetectEnabled:
                detected_frame = self.detectFace(frame)
                self.displayImage(detected_frame)
            else:
                self.displayImage(frame)
    # 是否使用外接摄像头
    def useExternalCamera(self):
        if self.extern_cameracheckBox.isChecked():
            self.isExternalCameraUsed = True
        else:
            self.isExternalCameraUsed = False
    #打开摄像头
    def startWebcam(self, status):
        if status:
            if self.isExternalCameraUsed:
                camID = 1
            else:
                camID = 0
            self.cap.open(camID)
            self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
            self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
            ret, frame = self.cap.read()
            if not ret:
                self.cap.release()
                self.opencamerapushButton_2.setChecked(False)
            else:
                self.opencamerapushButton_2.setText('关闭摄像头')
                self.openface_pushButton.setEnabled(True)
                self.timer.start(5)
        else:
            if self.cap.isOpened():
                if self.timer.isActive():
                    self.timer.stop()
                self.cap.release()
                self.face_showlabel.clear()
                self.face_showlabel.setText('摄像头未开启')
                self.openface_pushButton.setEnabled(False)
                self.opencamerapushButton_2.setText('打开摄像头')

        # 显示图像
    def displayImage(self, img):
        # BGR -> RGB
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        # default：The image is stored using 8-bit indexes into a colormap， for example：a gray image
        qformat = QImage.Format_Indexed8
        if len(img.shape) == 3:  # rows[0], cols[1], channels[2]
            if img.shape[2] == 4:
                # The image is stored using a 32-bit byte-ordered RGBA format (8-8-8-8)
                # A: alpha channel，不透明度参数。如果一个像素的alpha通道数值为0%，那它就是完全透明的
                qformat = QImage.Format_RGBA8888
            else:
                qformat = QImage.Format_RGB888

        # img.shape[1]：图像宽度width，img.shape[0]：图像高度height，img.shape[2]：图像通道数
        # QImage.__init__ (self, bytes data, int width, int height, int bytesPerLine, Format format)
        # 从内存缓冲流获取img数据构造QImage类
        # img.strides[0]：每行的字节数（width*3）,rgb为3，rgba为4
        # strides[0]为最外层(即一个二维数组所占的字节长度)，strides[1]为次外层（即一维数组所占字节长度），strides[2]为最内层（即一个元素所占字节长度）
        # 从里往外看，strides[2]为1个字节长度（uint8），strides[1]为3*1个字节长度（3即rgb 3个通道）
        # strides[0]为width*3个字节长度，width代表一行有几个像素

        outImage = QImage(img, img.shape[1], img.shape[0], img.strides[0], qformat)
        self.face_showlabel.setPixmap(QPixmap.fromImage(outImage))
        self.face_showlabel.setScaledContents(True)
if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = myfacewindow()
    window.show()
    sys.exit(app.exec_())

我们将自己要写的功能都可以在新建的这个Python文件去加，这样即使界面那个Py文件有所修改的话这里的话也不用担心被覆盖掉或者怎么样。:+1:,是不是非常的amazing!!!!!!.
